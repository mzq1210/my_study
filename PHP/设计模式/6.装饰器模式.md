/**
 * 需求：装饰器模式
 * 问题：装饰器模式例子中“人”类应该是Component还是ConcreteComponent类呢？
 * 解释：要灵活变通，如果只有一个ConcreteComponent类而没有抽象类Component，那Decorator抽象类可以是ConcreteComponent的一个子类
 *      同理，如果只有一个ConcreteComponent类，就没必要建一个单独的Decorator抽象类，可以把Decorator抽象类与ConcreteComponent合并成一个类
 * 总结：当需要执行特殊需求时，客户代码可以在运行时根据需要有选择的按顺序的使用装饰功能包装对象，这样可以有效地把类的核心职责与装饰功能分开，并且可以
 *      去除相关类中重复的装饰逻辑。但也要注意一种特殊情况：比如数据检测：加密数据肯定要再过滤功能之后，理想情况下，应该保证装饰器类之间互相独立。
 */

class PersonV3{
    public function show(){
        echo '小明</br>';
    }
}

class FineryV1 extends PersonV3 {
    public $person;

    //打扮
    public function setDecorate($person){
        $this->person = $person;
    }

    public function show(){
        if($this->person != null){
            //第一轮调用的是PersonV3的show()
            //之后调用的是子类的show()
            $this->person->show();
        }
    }
}

class minTshirtsV2 extends FineryV1{
    public function show(){
        //本次show()装饰之前先实现上一个show()装饰
        parent::show();
        echo '小T恤</br>';
    }
}

class TrouserV2 extends FineryV1{
    public function show(){
        parent::show();
        echo '裤子</br>';
    }
}

class Sneakers extends FineryV1{
    public function show(){
        parent::show();
        echo '鞋子</br>';
    }
}

$person = new PersonV3();
$ts = new minTshirtsV2();
$tr = new TrouserV2();
$sn = new Sneakers();

$ts->setDecorate($person); // 调用$ts->show() 输出：小明 小T恤
$sn->setDecorate($ts);     // 调用$sn->show() 输出：小明 小T恤 鞋子
$tr->setDecorate($sn);     // 调用$tr->show() 输出：小明 小T恤 鞋子 裤子
$tr->show();
